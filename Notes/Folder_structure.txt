/ (root)
├── assets/                (folder for images, fonts, etc.)
│   ├── images/            (subfolder for images)
│   │   ├── logo.png
│   │   └── background.jpg
│   └── fonts/             (subfolder for fonts)
│       └── custom-font.ttf
├── components/            (folder for reusable UI components)
│   ├── header.html        (header component)
│   ├── footer.html        (footer component)
│   └── nav.html           (navigation component)
├── pages/                 (folder for page-specific HTML files)
│   ├── about.html
│   ├── contact.html
│   └── gallery.html
├── styles/                (folder for CSS files)
│   ├── base.css           (base styles)
│   ├── layout.css         (layout-specific styles)
│   └── theme.css          (theme or color-specific styles)
├── scripts/               (folder for JavaScript files)
│   └── main.js            (main JavaScript file)
├── index.html             (your homepage HTML file, could be `homepage.html` or `index.html`)
└── styles.css             (if you still want a main stylesheet in the root, you can keep it here or merge with the styles/ folder)
